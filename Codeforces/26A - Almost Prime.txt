#include<bits/stdc++.h>
using namespace std;

int main() {
    int n,count=0;
    cin>>n;
    vector<int> prime(n+1,0);

    for(int i=2;i<=n;i++){
        if(prime[i]==0){
            for(int j=i;j<=n;j+=i)
            {
                prime[j]++;
            }
        }
    }
    for(int i=1;i<=n;i++){
        if(prime[i]==2)
            count++;
    }
    
    cout<<count<<"\n";
    return 0;
}
