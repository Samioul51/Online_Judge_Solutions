#include<bits/stdc++.h>
using namespace std;

#define ll long long int

void solve(){
    ll n,i;
    cin>>n;
    vector<ll> v(n);
    for(i=0;i<n;i++){
        cin>>v[i];
    }
    sort(v.begin(),v.end());
    ll j=0,k=1,count=0;
    while(k<n){
        if(v[j]<v[k]){
            count++;
            j++;
        }
        k++;
    }
    cout<<n-count<<endl;
}

int main(){
    ll tc;
    tc = 1;
    //cin>>tc;
    while(tc--){
        solve();
    }
    return 0;
}
