#include<bits/stdc++.h>
using namespace std;

#define ll long long int

void solve(){
    ll n,i;
    cin>>n;
    map<string,ll>m;
    deque<string>d;
    for(i=0;i<n;i++){
        string s;
        cin>>s;
        d.push_front(s);
        m[s]++;
    }
    while(!d.empty()){
        string top=d.front();
        d.pop_front();
        if(m[top]!=0){
            cout<<top<<endl;
            m[top]=0;
        }
    }   
}

int main(){
    ll tc;
    tc = 1;
    //cin>>tc;
    while(tc--){
        solve();
    }
    return 0;
}
