#include<bits/stdc++.h>
using namespace std;

#define ll long long int

void solve(){
    ll n;
    cin>>n;
    string s;
    cin>>s;
    ll i=0,j=n-1;
    while(i<n){
        if(s[i]=='B')
            break;
        i++;
    }
    while(j>=0){
        if(s[j]=='B')
            break;
        j--;
    }

    if(i>j)
        cout<<0<<endl;
    else
        cout<<j-i+1<<endl;
}

int main(){
    ll tc;
    tc = 1;
    cin>>tc;
    while(tc--){
        solve();
    }
    return 0;
}
