#include<bits/stdc++.h>
using namespace std;

#define ll long long int

void solve(){
    ll n,i;
    cin>>n;
    map<string,ll>m;
    for(i=0;i<n;i++){
        string s;
        cin>>s;
        if(!m.count(s)){
            cout<<"NO"<<endl;
            m[s]++;
        }
        else{
            cout<<"YES"<<endl;
            m[s]++;
        }
    }   
}

int main(){
    ll tc;
    tc = 1;
    //cin>>tc;
    while(tc--){
        solve();
    }
    return 0;
}
