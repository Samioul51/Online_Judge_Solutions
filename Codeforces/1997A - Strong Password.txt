#include<bits/stdc++.h>
using namespace std;

#define ll long long int

void solve(){
    string s;
    cin>>s;
    ll n=s.size(),i;
    vector<ll>v(26,0);
    for(i=0;i<n;i++){
        ll dummy=s[i]-'a';
        if(v[dummy]==0)
            v[dummy]=1;
    }
    char d;
    for(i=0;i<26;i++){
        if(v[i]==0){
            d='a'+i;
            break;
        }
    }
    ll dummy=-1;
    for(i=1;i<n;i++){
        if(s[i]==s[i-1]){
            dummy=i;
            break;
        }
    }
    if(dummy==-1)
        cout<<d<<s<<endl;
    else{
        for(i=0;i<dummy;i++){
            cout<<s[i];
        }
        cout<<d;
        for(i=dummy;i<n;i++){
            cout<<s[i];
        }
        cout<<endl;
    }
}

int main(){
    ll tc;
    tc = 1;
    cin>>tc;
    while(tc--){
        solve();
    }
    return 0;
}
